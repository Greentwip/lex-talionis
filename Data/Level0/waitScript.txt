if;self.unit.team == 'enemy'

        set_cursor;{unit};immediate

    elif;gameStateObj.mode['name'] == 'Lunatic' and self.unit.position[0] >= 6 and self.unit.position[1] <= 7 and self.unit.position != (6, 7)
        set_cursor;{unit};immediate
        disp_cursor;1
        wait;1000
        disp_cursor;0
        lose_game
    end
